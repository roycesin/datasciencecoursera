\HeaderA{comTraceObject}{Trace information about the passed object}{comTraceObject}
\keyword{programming}{comTraceObject}
\keyword{interface}{comTraceObject}
\begin{Description}\relax
This function will trace information about the COM object passed as parameter.

Use this function to find out what \code{rcom} thinks about the object.

Use this function for debugging purposes to find out why \code{rcom} fails.
\end{Description}
\begin{Usage}
\begin{verbatim}
comTraceObject(handle)
\end{verbatim}
\end{Usage}
\begin{Arguments}
\begin{ldescription}
\item[\code{handle}] COM object (class "COMObject") as returned by e.g.
\code{comCreateObject}
\end{ldescription}
\end{Arguments}
