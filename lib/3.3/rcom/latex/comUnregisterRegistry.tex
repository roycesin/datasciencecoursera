\HeaderA{comUnregisterRegistry}{Deregister the COM server from the registry}{comUnregisterRegistry}
\keyword{programming}{comUnregisterRegistry}
\keyword{interface}{comUnregisterRegistry}
\begin{Description}\relax
This function will deregister the COM server and its associated type library
from the registry/system.

Call this function before uninstalling the package to clean-up the registry.
\end{Description}
\begin{Usage}
\begin{verbatim}
comUnregisterRegistry()
\end{verbatim}
\end{Usage}

\end{Arguments}
