\HeaderA{comGetVersion}{Get rcom version information}{comGetVersion}
\keyword{programming}{comGetVersion}
\keyword{interface}{comGetVersion}
\begin{Description}\relax
This function returns the version of the currently installed \code{rcom}
package as a floating point number.

Use this function to find out which version you are using. The version must
be specified when reporting problems.
\end{Description}
\begin{Usage}
\begin{verbatim}
comGetVersion()
\end{verbatim}
\end{Usage}
\begin{Arguments}
\end{Arguments}
\begin{Value}
\code{comGetVersion} returns the version number in format \code{major.minor}.
\end{Value}
