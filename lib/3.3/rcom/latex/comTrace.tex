\HeaderA{comTrace}{Trace information about SEXP}{comTrace}
\keyword{programming}{comTrace}
\keyword{interface}{comTrace}
\begin{Description}\relax
This function will trace all passed parameters (variables, constants etc) to
the \R{} console.

Use this function to find out what \code{rcom} thinks about the parameters
(e.g. which data types are recognized).

Use this function for debugging purposes to find out why \code{rcom} fails.
\end{Description}
\begin{Usage}
\begin{verbatim}
comTrace(x)
\end{verbatim}
\end{Usage}
\begin{Arguments}
\begin{ldescription}
\item[\code{x}] expression (variable, constant) to trace
\end{ldescription}
\end{Arguments}
